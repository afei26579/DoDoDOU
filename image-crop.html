核心函数
/**
 * cropImage — 把图片按 transform 画到 canvas 再截取
 * @param {HTMLImageElement} img
 * @param {Object} t  — { x, y, scale, rotate }
 * @param {Object} frame — { w, h }  裁剪框尺寸(px, 视口坐标)
 * @returns {HTMLCanvasElement}
 */
function cropImage(img, t, frame) {
  const dpr = window.devicePixelRatio || 1;

  // 1. 创建输出 canvas，尺寸 = 裁剪框，考虑 DPR
  const canvas = document.createElement('canvas');
  canvas.width  = frame.w * dpr;
  canvas.height = frame.h * dpr;

  const ctx = canvas.getContext('2d');
  ctx.scale(dpr, dpr);

  // 2. 移动原点到裁剪框中心（画布中心）
  ctx.translate(frame.w / 2, frame.h / 2);

  // 3. 施加旋转
  ctx.rotate((t.rotate * Math.PI) / 180);

  // 4. 施加缩放
  ctx.scale(t.scale, t.scale);

  // 5. 把图片画进来，原点对齐图片中心
  //    t.x / t.y 是图片中心相对裁剪框中心的偏移
  ctx.drawImage(
    img,
    t.x / t.scale - img.naturalWidth  / 2,
    t.y / t.scale - img.naturalHeight / 2
  );

  // 6. 清理（可选，帮助 GC）
  // canvas.width = 0; canvas.height = 0;

  return canvas;
}
